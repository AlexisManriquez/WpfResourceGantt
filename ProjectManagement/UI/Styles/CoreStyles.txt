                                                Content="{Binding Tag, RelativeSource={RelativeSource TemplatedParent}}"
                                                Visibility="Visible">
                                    <ContentControl.Resources>
                                        <DataTemplate DataType="{x:Type Geometry}">
                                            <Path Data="{Binding}" Fill="{Binding Foreground, RelativeSource={RelativeSource AncestorType=Button}}" Stretch="Uniform"/>

                                            <Image Source="{Binding}" Stretch="Uniform"/>
                                        </DataTemplate>
                                    </ContentControl.Resources>
                                </ContentControl>

                                <!-- Active icon: shown when button is selected (swapped via trigger).
                                     Bound to the RibbonProperties.ActiveIcon attached property.
                                     For Geometry icons this stays null and StandardIcon handles it via Foreground. -->
                                <ContentControl x:Name="ActiveIcon"
                                                Content="{Binding Path=(ui:RibbonProperties.ActiveIcon), RelativeSource={RelativeSource TemplatedParent}}"
                                                Visibility="Collapsed">
                                    <ContentControl.Resources>
                                        <DataTemplate DataType="{x:Type DrawingImage}">
                                            <Image Source="{Binding}" Stretch="Uniform"/>
                                        </DataTemplate>
                                    </ContentControl.Resources>
                                </ContentControl>
                            </Grid>

                            <!-- Text (Bottom) -->
                            <TextBlock x:Name="textBlock"
                                       Grid.Row="1"
                                       Text="{TemplateBinding Content}"
                                       Foreground="{TemplateBinding Foreground}"
                                       FontSize="11"
                                       FontWeight="SemiBold"
                                       TextAlignment="Center"
                                       TextWrapping="Wrap"
                                       MaxWidth="80"
                                       HorizontalAlignment="Center"/>
                        </Grid>
                    </Border>
                    <ControlTemplate.Triggers>
                        <DataTrigger Binding="{Binding DataContext.IsRibbonExpanded, RelativeSource={RelativeSource AncestorType=Window}}" Value="False">
                            <Setter TargetName="IconContainer" Property="Visibility" Value="Collapsed"/>
                            <Setter TargetName="textBlock" Property="Margin" Value="0,2"/>
                            <Setter TargetName="textBlock" Property="FontSize" Value="12"/>
                        </DataTrigger>
                        <!-- Active View State Logic -->
                        <DataTrigger Value="True">
                            <DataTrigger.Binding>
                                <MultiBinding Converter="{StaticResource EqualityToBoolMulti}">
                                    <Binding Path="DataContext.CurrentView" RelativeSource="{RelativeSource AncestorType=UserControl}"/>
                                    <Binding Path="CommandParameter" RelativeSource="{RelativeSource Self}"/>
                                </MultiBinding>
                            </DataTrigger.Binding>
                            <Setter TargetName="border" Property="Background" Value="{StaticResource TacticalGoldBrush}"/>
                            <Setter Property="Foreground" Value="{StaticResource TacticalTextInverseBrush}"/>
                            <Setter TargetName="textBlock" Property="FontWeight" Value="Bold"/>

                            <!-- Swap to active icon variant (if one is provided via attached property) -->
                            <Setter TargetName="StandardIcon" Property="Visibility" Value="Collapsed"/>
                            <Setter TargetName="ActiveIcon" Property="Visibility" Value="Visible"/>
                        </DataTrigger>

                        <!-- Hover State -->
                        <MultiDataTrigger>
                            <MultiDataTrigger.Conditions>
                                <Condition Binding="{Binding IsMouseOver, RelativeSource={RelativeSource Self}}" Value="True"/>
                                <Condition Value="False">
                                    <Condition.Binding>
                                        <MultiBinding Converter="{StaticResource EqualityToBoolMulti}">
                                            <Binding Path="DataContext.CurrentView" RelativeSource="{RelativeSource AncestorType=UserControl}"/>
                                            <Binding Path="CommandParameter" RelativeSource="{RelativeSource Self}"/>
                                        </MultiBinding>
